ls - показывает содержимое папки (файлы и другие папки)
ls название_директрии{py33} - показывает все файлы и папки внутри папки py33
ls -a = показывает скрытые файлы и папки
ls -l = показывает дополнительную информацию о файлах и директориях
ls - la = показывает скрытые файлы и директории с подробной инфо/кобинация -a / -l

cd =  переход в доманнюю территорию
cd ~ =  тоже переход в доманнюю территорию
cd test = переход в папку test, если такой директории нет, то выйдет ошибка
cd .. =  переход на директорию выше, на пред. 
cd ../.. = переход на 2 директории выше
cd - = cntr + z = выполнение последной команды 

mkdir test = создает папку test в текущей директории
mkdir test/test2 = создает папку test2 внутри папки test, если папка test существует, зонст ошибка
mkdir test test2 = создает папку test и tes2

touch file.txt = создает файл 'file' с разрешением 'txt'/ если не указать формат, то выходит просто в виде текста
touch file1.txt file2.txt = создает 'file1.txt'и 'file2.txt'

rm test.txt = удаляет файл
rmdir test = удаляет пустую директорию test
rm - rf = удаляет всё: и файлы, и директории с содержимым, пустые директории. Рекурсивное удаление(полное удаление)

mv test.txt test = перемещает файл test.txt в директорию test
mv .. = перемещает файл на директорию назад
mv test1.txt test2.txt = переименовывает test1.txt на test2.txt

pwd(print working directory) = показывает абсолютный путь (начиная с корневой папки/) до текущей директории

history = показывает все команды, которые мы вводили в данный терминал, можно с ограничинем
hisrory 10 = показывает 10 последних введеных команд из истории команд 
history | grep test = показывает команды из истории команд содержавшие слово 'test'

nano test.txt = открывает файл test.txt в текстовом редакторе nano 

cat test.txt = выводит содержимое файла test.txt в терминал

sudo(super user do) = даёт права суперюзера

apt = системный установщик пакетов, на МАКах нет, чаще использует для 'sudo'(ubuntu)
brew = установщик пактов (macos)

cntr + x + y +enter = сохранить
